code Install Widget

Add this snippet to your website to show the feedback widget:

<script src="https://painsolver.vercel.app/sdk/painsolver.js"></script>
<script>
  PainSolver("init", {
    apiBaseUrl: "https://painsolver.vercel.app",
    boardToken: "YOUR_BOARD_TOKEN"
  });

  // Identify logged-in users (optional but recommended)
  PainSolver("identify", {
    user: { email: "user@example.com", name: "User Name" },
    company: { name: "Company", monthlySpend: 100 },
    hash: "HMAC_FROM_YOUR_BACKEND"
  });

  PainSolver("render", { selector: "#feedback-widget" });
</script>
<div id="feedback-widget"></div>
